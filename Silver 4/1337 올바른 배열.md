### 올바른 배열

<br>
문제 https://www.acmicpc.net/problem/1337
<br>
<br>


```C
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>
int main() {

	int n;
	int i, j, k, l;
	int number_of_one = -1;
	int cnt;
	int* arr;

	scanf("%d", &n);
	arr = (int*)malloc(sizeof(int) * n);

	for (i = 0; i < n; i++) {
		scanf("%d", &arr[i]);
	}

	for (i = 0; i < n; i++) {
		for (j = 4; j >= 0; j--) {
			cnt = 0;
			if (arr[i] - j <= -1)
				continue;
			for (k = 0; k < 5; k++) {

				for (l = 0; l < n; l++) {
					if (arr[i] - j + k == arr[l])
						cnt++;
				}
				if (number_of_one < cnt)
					number_of_one = cnt;
			}
		}
	}

	printf("%d", 5 - number_of_one);
}
```
