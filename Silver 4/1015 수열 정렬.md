###  수열정렬

<br>
문제 https://www.acmicpc.net/problem/1015
<br>
<br>


```C
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <math.h>
int main() { 
	int p[50], a[50], n; 
	scanf("%d", &n); 
	for (int i = 0; i < n; i++) {
		scanf("%d", &a[i]);
		p[i] = i;
	}
	for (int i = 1; i < n; i++) { 
		int j = i; 
		int v = p[i]; 
		while (j > 0 && a[p[j - 1]] > a[v]) { 
			p[j] = p[j - 1];
			j--; 
		} 
		p[j] = v; 
	} 
	for (int i = 0; i < n; i++) {
		a[p[i]] = i;
	}
	for (int i = 0; i < n; i++) {
		printf("%d ", a[i]);
	}
}
```
