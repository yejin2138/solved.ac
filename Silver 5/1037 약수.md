
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>

int main() {
	int  num;
	long long int min, max;
	scanf("%d", &num);

	int* val = (int*)malloc(sizeof(int) * num);

	for (int i = 0; i < num; i++) {
		scanf("%d", &val[i]);
	}
	min = val[0];
	max = val[0];
	for (int i = 1; i < num; i++) {
		
		if (min > val[i]) {
			min = val[i];
		}
		if (max < val[i]) {
			max = val[i];
		}
		
	}
	
	printf("%d\n", min*max);
}
