```C
#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>

int main() {
    int row, col;
    char arr[50][50];
    int min = 32;

    scanf("%d %d", &row, &col);
    for (int i = 0; i < row; i++) {
        scanf("%s", arr[i]);
    }
    for (int i = 0; i < row - 7; i++) {
        for (int j = 0; j < col - 7; j++) {
            int count = 0;
            for (int ii = i; ii < i + 8; ii++) {
                for (int jj = j; jj < j + 8; jj++) {
                    if ((ii + jj) % 2 == 1) {
                        if (arr[ii][jj]!='B')
                        count++;
                    }
                    if ((ii + jj) % 2 == 0) {
                        if (arr[ii][jj] != 'W')
                            count++;
                    }
                }
            }
            if (count > (64 - count)) {
                count = 64 - count;
            }
            if (min > count) {
                min = count;
            }
        }
    }
    printf("%d", min);
}
```
