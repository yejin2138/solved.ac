#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
int main() {
	int N, M, num, val;
	unsigned long long result=1;
	scanf("%d", &num);
	
	for (int i = 0; i < num; i++) {
		scanf("%d %d", &N,&M);
		result = 1;
		if (N > (M - N)) {
			val = N;
		}
		else {
			val = (M-N);
		}
		for (int j = val + 1; j <= M; j++) {
			result *= j;
		}
		for (int j = 1; j <= (M-val); j++) {
			result /= j;
		}
		printf("%d\n", result);
	}
}
