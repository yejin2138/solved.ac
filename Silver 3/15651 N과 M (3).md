### N과 M (3)

<br>
문제 https://www.acmicpc.net/problem/15651
<br>
<br>

```C
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>

int arr[7] = { 0, };
void pick( int n, int m, int k) {
	for (int i = 1; i <= n; i++) {
		arr[k] = i;
		if (m - 1 == k) {
			for (int k = 0; k < m; k++)
				printf("%d ", arr[k]);
			printf("\n");
		}
		else pick(n, m, k + 1);
	}
}

int main(void) {
	int n, m;
	scanf("%d %d", &n, &m);

	pick(n,m,0);
}
```
