### N과 M (4)

<br>
문제 https://www.acmicpc.net/problem/15652
<br>
<br>

```C
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>

int arr[8] = { 0, };

void pick(int N, int M, int d) {
	for (int i = 1, k; i <= N; i++) {
		if (d == 0) {
			arr[d] = i;
		}
		else {
			for (k = 0; k < d; k++) {
				if (arr[k] > i) {
					break;
				}
			}
			if (k == d) {
				arr[d] = i;
			}
			else {
				continue;
			}
		}
	
		if (M == d + 1) {
			for (int n = 0; n < M; n++) {
				printf("%d ", arr[n]);
			}

			printf("\n");
		}
		else {
			pick(N, M, d + 1);
		}
	}
}

int main() {
	int N, M;
	scanf("%d %d", &N, &M);

	pick(N, M, 0);

	return 0;
}

```
